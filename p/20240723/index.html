    public static boolean isEmpty(Object obj) {
        if (null == obj) {
            return true;
        }

        if (obj instanceof CharSequence) {
            return StrUtil.isEmpty((CharSequence) obj);
        } else if (obj instanceof Map) {
            return MapUtil.isEmpty((Map) obj);
        } else if (obj instanceof Iterable) {
            return IterUtil.isEmpty((Iterable) obj);
        } else if (obj instanceof Iterator) {
            return IterUtil.isEmpty((Iterator) obj);
        } else if (ArrayUtil.isArray(obj)) {
            return ArrayUtil.isEmpty(obj);
        }

        return false;
    }

equal比较

使用("gcx_blood").equals(param.getObsvCode())而不是param.getObsvCode().equals()

这样写可以避免空指针异常，如果param.getObsvCode()返回null。调用equal()会抛异常，而使用("gcx_blood").equals(null) 不会抛出异常，而是直接返回 false。

	/**
	 * 格式化日期时间为yyyy-MM-dd HH:mm:ss格式
	 *
	 * @param time {@link LocalDateTime}
	 * @return 格式化后的字符串

    /**
     * 将字符串转为LocalDateTime
     *
     * @param dateTimeString
     * @return java.time.LocalDateTime
     * @author 刘书良
     * 2025/1/9 9:34

    public static LocalDateTime stringToLocalDateTime(String dateTimeString) {
        LocalDateTime localDateTime = LocalDateTimeUtil.parse(dateTimeString, DatePattern.NORM_DATETIME_PATTERN);
        return localDateTime;
    }
